\lstset{
    commentstyle=\color{codegreen},
    keywordstyle=\color{blue},
    numberstyle=\tiny\color{codegray},
    stringstyle=\color{codered},
    basicstyle=\ttfamily,
    breaklines=true,
    keepspaces=true,

    numbersep=8pt,
    frame=single,
    rulecolor=\color{framegray},
